    **Replace dryStep() body** (lines 1409-1449) with LUT-driven logic. The new dryStep:
    1. Iterates all pixels where `wetAlpha[i] >= DRY_ALPHA_THRESHOLD`
    2. Advances `dryPos[i]` by `drySpeed` (clamped to LUT_SIZE)
    3. Computes current LUT fraction: `frac = dryLUT[Math.floor(dryPos[i])]`

    7. Canvas alpha: `sa = drain / 800` (same normalization as current code)

    Full replacement dryStep body:
    ```javascript
    function dryStep() {
      const id = X.getImageData(0, 0, W, H);
      const d = id.data;
      let changed = false;

      for (let i = 0; i < W * H; i++) {
        if (wetAlpha[i] < DRY_ALPHA_THRESHOLD) continue;

        const prevPos = dryPos[i];
        dryPos[i] = Math.min(LUT_SIZE, dryPos[i] + drySpeed);

        // Fully dry: transfer everything remaining
        if (dryPos[i] >= LUT_SIZE) {
          const sa = Math.min(1, wetAlpha[i] / 800);
          const pi = i * 4;
          const ma = d[pi + 3] / 255;
          if (sa > 0.001) {
            const oa = Math.min(1, ma + sa * (1 - ma));
            const bt = sa / Math.max(0.001, oa);
            d[pi]     = Math.round(clamp(lerp(d[pi],     wetR[i], bt), 0, 255));
            d[pi + 1] = Math.round(clamp(lerp(d[pi + 1], wetG[i], bt), 0, 255));
            d[pi + 2] = Math.round(clamp(lerp(d[pi + 2], wetB[i], bt), 0, 255));
            d[pi + 3] = Math.round(clamp(oa * 255, 0, 255));
            changed = true;
          }
          wetAlpha[i] = 0; wetness[i] = 0;
          wetR[i] = 0; wetG[i] = 0; wetB[i] = 0;
          dryPos[i] = 0;
          continue;
        }

        // LUT-driven fractional transfer
        const frac = dryLUT[Math.floor(dryPos[i])];
        const prevFrac = dryLUT[Math.floor(prevPos)];
        const df = frac - prevFrac;
        if (df < 0.0001) continue;

        const drain = wetAlpha[i] * df;
        let sa = drain / 800;

        // Paper texture modulation (unchanged)
        if (paperHeight) {
          const ph = paperHeight[i];
          sa *= clamp(1.4 - ph * 0.8, 0.3, 1.4);
        }

        const pi = i * 4;
        const ma = d[pi + 3] / 255;

        if (sa > 0.001) {
          const oa = Math.min(1, ma + sa * (1 - ma));
          const bt = sa / Math.max(0.001, oa);
          d[pi]     = Math.round(clamp(lerp(d[pi],     wetR[i], bt), 0, 255));
          d[pi + 1] = Math.round(clamp(lerp(d[pi + 1], wetG[i], bt), 0, 255));
          d[pi + 2] = Math.round(clamp(lerp(d[pi + 2], wetB[i], bt), 0, 255));
          d[pi + 3] = Math.round(clamp(oa * 255, 0, 255));
          changed = true;
        }

        wetAlpha[i] -= drain;
        wetness[i] = Math.max(0, wetness[i] * (1 - df));

        if (wetAlpha[i] < DRY_ALPHA_THRESHOLD) {
          wetAlpha[i] = 0; wetness[i] = 0;
          wetR[i] = 0; wetG[i] = 0; wetB[i] = 0;
          dryPos[i] = 0;
        }
      }

      if (changed) X.putImageData(id, 0, 0);
    }
    ```
