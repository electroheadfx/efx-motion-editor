<objective>
Implement the select tool, post-process FX application workflow with per-frame rendering, sequence overlay toggle, and wire the preview renderer to use solid backgrounds.

Purpose: This is the core UX change from "draw with style" to "draw flat, select, apply FX" (per D-01, D-02, D-05, D-06). Users draw in flat mode always, then use the select tool to pick strokes and apply FX styles. FX application calls renderFrameFx() which renders ALL FX-applied strokes on the frame together on one p5.brush canvas for spectral mixing (PAINT-06). The frame-level cache is stored in paintStore. The preview renderer uses renderPaintFrameWithBg() for solid background compositing. Also implements D-13/D-14 sequence overlay toggle.

CRITICAL: FX application renders ALL FX strokes on the frame together (not just the newly-selected ones). Undo/rollback of a single stroke's FX triggers full-frame re-render of all remaining FX strokes. This is required for spectral mixing correctness.

Output: PaintOverlay with select tool and FX application, previewRenderer using renderPaintFrameWithBg, PaintProperties with overlay toggle.
</objective>

```typescript
function handleSelectPointerDown(e: PointerEvent) {
  if (e.button !== 0) return;
  const point = getProjectPoint(e);
  const layerId = getSelectedPaintLayerId();
  if (!layerId) return;

  const frame = timelineStore.currentFrame.peek();
  const paintFrame = paintStore.getFrame(layerId, frame);
  if (!paintFrame) return;

  // Find the topmost stroke that the click hits (reverse order = top first)
  const hitStrokeId = findStrokeAtPoint(paintFrame, point.x, point.y);

  if (hitStrokeId) {
    // Toggle selection (Cmd/Ctrl for multi-select, plain click for single)
    if (e.metaKey || e.ctrlKey) {
      paintStore.toggleStrokeSelection(hitStrokeId);
    } else {
      paintStore.clearSelection();
      paintStore.selectStroke(hitStrokeId);
    }
  } else {
    // Click on empty space: clear selection
    paintStore.clearSelection();
  }

  e.preventDefault();
}
```

**5. Implement hit testing for stroke selection:**

Add a helper function to find which stroke a point falls on:

```typescript
function findStrokeAtPoint(
  paintFrame: PaintFrame,
  x: number,
  y: number,
): string | null {
  // Walk elements in reverse order (topmost first)
  for (let i = paintFrame.elements.length - 1; i >= 0; i--) {
    const el = paintFrame.elements[i];
    if (el.tool !== 'brush') continue;
    const stroke = el as PaintStroke;

    // Check if point is within stroke bounding box + padding
    const pad = stroke.size / 2 + 5; // half brush size + 5px tolerance
    let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
    for (const [px, py] of stroke.points) {
      if (px < minX) minX = px;
      if (py < minY) minY = py;
      if (px > maxX) maxX = px;
      if (py > maxY) maxY = py;
    }

    if (x >= minX - pad && x <= maxX + pad && y >= minY - pad && y <= maxY + pad) {
      // Fine check: distance to any point in the stroke
      for (const [px, py] of stroke.points) {
        const dx = x - px;
        const dy = y - py;
        if (dx * dx + dy * dy <= pad * pad) {
          return stroke.id;
        }
      }
    }
  }
  return null;
}
```

**8. Add FX application via useEffect watching brushStyle + selectedStrokeIds:**

When the user has strokes selected and changes the brush style in PaintProperties, apply FX to those strokes. CRITICAL: This triggers a FULL FRAME re-render of ALL FX strokes (not just newly selected ones) for spectral mixing.

```typescript
useEffect(() => {
  const style = paintStore.brushStyle.value;
  const selected = paintStore.selectedStrokeIds.peek();
  if (selected.size === 0) return;

  const layerId = getSelectedPaintLayerId();
  if (!layerId) return;
  const frame = timelineStore.currentFrame.peek();
  const paintFrame = paintStore.getFrame(layerId, frame);
  if (!paintFrame) return;

  const projW = projectStore.width.peek();
  const projH = projectStore.height.peek();

  if (style === 'flat') {
    // Rollback selected strokes to flat (per D-10)
    let changed = false;
    for (const el of paintFrame.elements) {
      if (el.tool !== 'brush') continue;
      if (!selected.has(el.id)) continue;
      const stroke = el as PaintStroke;
      if (stroke.fxState === 'fx-applied' || stroke.fxState === 'flattened') {
        stroke.brushStyle = 'flat';
        stroke.brushParams = undefined;
        stroke.fxState = 'flat';
        changed = true;
      }
    }
    if (changed) {
      // Invalidate and re-render frame cache (remaining FX strokes still need spectral mixing)
      paintStore.invalidateFrameFxCache(layerId, frame);
      reRenderFrameFx(paintFrame, layerId, frame, projW, projH);
      paintStore.markDirty(layerId, frame);
      paintStore.paintVersion.value++;
    }
    return;
  }

  // Apply FX style to selected strokes (per D-08)
  let changed = false;
  for (const el of paintFrame.elements) {
    if (el.tool !== 'brush') continue;
    if (!selected.has(el.id)) continue;
    const stroke = el as PaintStroke;

    stroke.brushStyle = style;
    stroke.brushParams = { ...paintStore.brushFxParams.peek() };
    stroke.fxState = 'fx-applied';
    changed = true;
  }

  if (changed) {
    // Re-render ALL FX strokes on the frame together (spectral mixing!)
    reRenderFrameFx(paintFrame, layerId, frame, projW, projH);
    paintStore.markDirty(layerId, frame);
    paintStore.paintVersion.value++;
  }
}, [paintStore.brushStyle.value]);
```

**9. Add the frame-level re-render helper:**

```typescript
/**
 * Re-render all FX-applied strokes on a frame together via p5.brush.
 * All FX strokes go through renderFrameFx() on one canvas for spectral mixing.
 * Result is cached in paintStore.frameFxCache.
 */
function reRenderFrameFx(
  paintFrame: PaintFrame,
  layerId: string,
  frame: number,
  width: number,
  height: number,
): void {
  // Collect ALL brush strokes from the frame (renderFrameFx will filter to fx-applied)
  const brushStrokes = paintFrame.elements.filter(
    (el) => el.tool === 'brush'
  ) as PaintStroke[];

  const cached = renderFrameFx(brushStrokes, width, height);
  if (cached) {
    paintStore.setFrameFxCache(layerId, frame, cached);
  } else {
    // No FX strokes remain -- clear cache
    paintStore.invalidateFrameFxCache(layerId, frame);
  }
}
```
