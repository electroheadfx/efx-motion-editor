function getSelectedContentLayer() {
  const layerId = layerStore.selectedLayerId.value;
  if (!layerId) return null;  // <-- returns null, cascading to empty keyframes
  const seqs = sequenceStore.sequences.value;
  for (const seq of seqs) {
    if (seq.kind === 'fx') continue;
    const layer = seq.layers.find(l => l.id === layerId);
    if (layer && !layer.isBase && !isFxLayer(layer)) return { layer, sequenceId: seq.id };
  }
  return null;
}

The logic: When the user clicks on a content track area (either the header or the body), we check if the currently selected layer is an FX layer. If so, we clear it (preserving the original intent of "Delete targets the sequence"). If the current selection is a content layer, we leave it alone — this preserves the `selectedLayerId` signal, which keeps `activeLayerKeyframes` populated, which keeps diamonds rendered and the PropertiesPanel keyframe UI visible.

Extract the repeated conditional-clear logic into a private helper method to avoid duplication:

```typescript
/** Clear layer selection only if current selection is an FX layer.
 *  Preserves content layer selection so keyframe diamonds stay visible. */
private clearFxLayerSelection(): void {
  const currentLayerId = layerStore.selectedLayerId.peek();
  if (!currentLayerId) return;
  const allSeqs = sequenceStore.sequences.peek();
  const currentLayer = allSeqs.flatMap(s => s.layers).find(l => l.id === currentLayerId);
  if (currentLayer && isFxLayer(currentLayer)) {
    layerStore.setSelected(null);
    uiStore.selectLayer(null);
  }
}
```

Then both fix sites simply call `this.clearFxLayerSelection()` instead of the unconditional `layerStore.setSelected(null); uiStore.selectLayer(null);`.
