```typescript
export const canvasStore = {
  zoom: Signal<number>,          // current zoom (0.1-4.0)
  panX: Signal<number>,          // pan X offset
  panY: Signal<number>,          // pan Y offset
  zoomPercent: ReadonlySignal<number>,  // computed: Math.round(zoom * 100)
  isAtMinZoom: ReadonlySignal<boolean>, // computed: zoom <= 0.1
  isAtMaxZoom: ReadonlySignal<boolean>, // computed: zoom >= 4.0

  zoomIn(): void,              // snap to next preset up, center-anchored
  zoomOut(): void,             // snap to next preset down, center-anchored
  setSmoothZoom(newZoom: number, cursorX: number, cursorY: number): void,
  fitToWindow(): void,         // calculate fit scale, reset pan to centered
  updateContainerSize(w: number, h: number): void,
  reset(): void,
};
```

**shortcuts.ts:**

1. **Import canvasStore** at top: `import {canvasStore} from '../stores/canvasStore';`

2. **Add 3 new shortcut bindings** inside the `tinykeys(window, { ... })` object in `mountShortcuts()`, after the theme cycle entry:

   ```typescript
   // Canvas zoom (ZOOM-01, ZOOM-03)
   '$mod+Equal': (e: KeyboardEvent) => {
     if (shouldSuppressShortcut(e)) return;
     e.preventDefault();
     canvasStore.zoomIn();
   },
   '$mod+Minus': (e: KeyboardEvent) => {
     if (shouldSuppressShortcut(e)) return;
     e.preventDefault();
     canvasStore.zoomOut();
   },
   '$mod+Digit0': (e: KeyboardEvent) => {
     if (shouldSuppressShortcut(e)) return;
     e.preventDefault();
     canvasStore.fitToWindow();
   },
   ```

Per research: tinykeys `$mod` = Cmd on macOS, Ctrl on Windows/Linux. `Equal` = the =/+ key. `Minus` = -/_ key. `Digit0` = 0 key. Tauri v2 `zoomHotkeysEnabled` defaults to false, so these keys are free. Still call `e.preventDefault()` to be safe.

Per research (Pitfall 2): Cmd+0 is NOT a native accelerator in the existing macOS menu (only App + Edit submenus), so tinykeys will receive it.

**ShortcutsOverlay.tsx:**

3. **Add a 'Canvas' shortcut group** to the `SHORTCUT_GROUPS` array, placed after the 'Navigation' group:


   Note: `\u2318` is the command symbol, `\u2212` is the minus sign. This matches the style of existing entries (e.g., '\u2318S' for Save, '\u21E7\u2318Z' for Redo).
