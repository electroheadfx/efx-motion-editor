   ```typescript
     setMenuOpen(false);

     const fxLayer: Layer = {

     uiStore.selectLayer(layerId);
   ```

2. In the `renderCurrent()` function AND in the render effect, after the existing `renderer.renderFrame(layers, localFrame, frames, fps)` call, add FX sequence compositing:

   ```typescript
   // After rendering content layers, composite FX sequences on top
   const globalFrame = timelineStore.currentFrame.peek(); // (or .value in the effect)
   const fxSequences = sequenceStore.getFxSequences();
   for (const fxSeq of fxSequences) {
     // Skip FX sequences outside their temporal range
     if (fxSeq.inFrame != null && globalFrame < fxSeq.inFrame) continue;
     if (fxSeq.outFrame != null && globalFrame >= fxSeq.outFrame) continue;
     // Render FX layers on top of current content
     const fxLayers = fxSeq.layers.filter((l) => l.visible);
     if (fxLayers.length > 0) {
       renderer.renderFrame(fxLayers, localFrame, frames, fps);
     }
   }
   ```


   In Preview.tsx, the content render call stays as-is (clearCanvas defaults to true). FX overlay calls pass `false`:
