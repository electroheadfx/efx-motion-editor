```rust
/// Check if a file path exists on disk.
/// Uses std::path::Path directly — not restricted by Tauri FS scope.
#[command]
pub fn path_exists(file_path: String) -> bool {
    std::path::Path::new(&file_path).exists()
}
```

2. In `Application/src-tauri/src/lib.rs`, register the new command in the `generate_handler!` macro — add `project::path_exists` to the list.

3. In `Application/src/lib/ipc.ts`, add the TypeScript wrapper:
```typescript
export async function pathExists(filePath: string): Promise<Result<boolean>> {
  return safeInvoke<boolean>('path_exists', { filePath });
}
```

This bypasses the FS plugin scope restriction entirely by using Rust's std::path::Path::exists() which has no scope limitations.
