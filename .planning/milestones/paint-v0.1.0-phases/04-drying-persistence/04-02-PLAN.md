    ```javascript
    // ============================================================
    //  STROKE SERIALIZATION (STROKE-03, D-04/D-05/D-06)
    //  JSON: canvas settings header + stroke array
    //  Points as compact arrays [x, y, p, tx, ty, tw, spd] per D-05/Pitfall 7
    // ============================================================

    function serializeProject() {
      return {
        version: 1,
        canvas: {
          width: W,
          height: H,
          bgMode: bgMode,
          paperGrain: currentPaperKey,
          embossStrength: embossStrength,
          wetPaper: wetPaper,
        },
        strokes: allActions.map(s => ({
          tool: s.tool,
          points: (s.points || s.pts).map(p => [
            Math.round(p.x * 100) / 100,
            Math.round(p.y * 100) / 100,
            Math.round((p.p || 0) * 1000) / 1000,
            p.tx || 0,
            p.ty || 0,
            p.tw || 0,
            Math.round((p.spd || 0) * 100) / 100,
          ]),
          color: s.color,
          params: s.params || s.opts,
          timestamp: s.timestamp,
        })),
      };
    }

    function loadProject(json) {
      const data = typeof json === 'string' ? JSON.parse(json) : json;

      // 1. Restore canvas settings (per D-06)
      if (data.canvas) {
        bgMode = data.canvas.bgMode || 'transparent';
        // Update background button UI
        document.querySelectorAll('.bgb').forEach(x => x.classList.remove('active'));
        const bgBtn = document.querySelector('[data-bg="' + bgMode + '"]');
        if (bgBtn) bgBtn.classList.add('active');
        wrap.classList.toggle('checker', bgMode === 'transparent');

        // Restore paper grain
        if (data.canvas.paperGrain) {
          document.querySelectorAll('.pgb').forEach(x => x.classList.remove('active'));
          const pgBtn = document.querySelector('.pgb[data-paper="' + data.canvas.paperGrain + '"]');
          if (pgBtn) pgBtn.classList.add('active');
          setPaperPhysics(data.canvas.paperGrain);
        }

        // Restore emboss strength
        if (data.canvas.embossStrength != null) {
          embossStrength = data.canvas.embossStrength;
          document.querySelectorAll('.grain-btn').forEach(x => {
            x.classList.toggle('active', +x.dataset.grain === embossStrength);
          });
        }

        // Restore wet paper mode
        if (data.canvas.wetPaper != null) {
          wetPaper = data.canvas.wetPaper;
          directRender = !wetPaper;
          document.querySelectorAll('.paper-btn').forEach(x => {
            x.classList.toggle('active', (x.dataset.paper === 'wet') === wetPaper);
          });
        }
      }

      // 2. Convert compact point arrays back to objects
      allActions = data.strokes.map(s => ({
        tool: s.tool,
        points: s.points.map(p => ({
          x: p[0], y: p[1], p: p[2],
          tx: p[3], ty: p[4], tw: p[5], spd: p[6],
        })),
        color: s.color,
        params: s.params,
        timestamp: s.timestamp,
      }));

      // 3. Clear undo stack (loaded state is new baseline)
      undoStack = [];

      // 4. Replay with physics fast-forward (per D-07/D-10)
      replayWithPhysics(allActions);
    }

    function replayWithPhysics(strokes) {
      // Pause live physics to prevent non-deterministic interference (Pitfall 3)
      clearInterval(physicsInterval);

      drawBg();
      clearWetLayer();
      directRender = true;

      for (let i = 0; i < strokes.length; i++) {
        const s = strokes[i];

        // Fast-forward physics for elapsed time since previous stroke (per D-07)
        if (i > 0 && s.timestamp && strokes[i - 1].timestamp) {
          const elapsed = s.timestamp - strokes[i - 1].timestamp;
          const ticks = Math.floor(elapsed / 100);  // 100ms per physics tick
          const cappedTicks = Math.min(ticks, 500);  // cap at 50 seconds
          for (let t = 0; t < cappedTicks; t++) {
            physicsStep();
          }
        }

        // Dispatch stroke using same logic as redrawAll
        const pts = s.points || s.pts;
        const opts = s.params || s.opts;
        const color = s.color;
        if (s.tool === 'paint') renderPaintStroke(pts, color, opts);
        else if (s.tool === 'liquify') {
          const r = opts.size || opts.radius || 24;
          const w = (opts.waterAmount != null ? opts.waterAmount : (opts.wetness || 35));
          applyLiquifyChunk(pts, r, w / 100 * 0.8 + 0.1);
        }
        else if (s.tool === 'erase') applyEraseStroke(pts, opts);
        else if (s.tool === 'water') applyWaterStroke(pts, opts);
        else if (s.tool === 'smear') applySmearChunk(pts, opts);
        else if (s.tool === 'blend') applyBlendStroke(pts, opts);
        else if (s.tool === 'blow') applyBlowChunk(pts, opts);
        else if (s.tool === 'wet') applyWetChunk(pts, opts);
        else if (s.tool === 'dry') applyDryChunk(pts, opts);
      }

      // Final physics pass: run a few ticks to settle any remaining wet paint
      for (let t = 0; t < 10; t++) physicsStep();

      directRender = false;

      // Resume live physics
      physicsInterval = setInterval(physicsStep, 100);
    }

    // ============================================================
    //  SAVE / LOAD — file dialog (D-09)
    //  Save downloads .efxpaint.json, Load opens file picker
    // ============================================================

    function saveProject() {
      const data = serializeProject();
      const json = JSON.stringify(data);
      const blob = new Blob([json], { type: 'application/json' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'painting.efxpaint.json';
      a.click();
      URL.revokeObjectURL(url);
    }

    function loadProjectFromFile() {
      const input = document.createElement('input');
      input.type = 'file';
      input.accept = '.json,.efxpaint.json';
      input.onchange = (e) => {
        const file = e.target.files[0];
        if (!file) return;
        const reader = new FileReader();
        reader.onload = (ev) => {
          try {
            loadProject(ev.target.result);
          } catch (err) {
            console.error('Failed to load project:', err);
            alert('Failed to load painting: ' + err.message);
          }
        };
        reader.readAsText(file);
      };
      input.click();
    }
    ```
