**1. applyEraseChunk(pts, opts) -- real-time tool**

Reduces wetAlpha and wetness under the brush. Uses heightmap mask but NOT brush grain.

```javascript
function applyEraseChunk(pts, opts) {
  if (pts.length < 2) return;
  const { size, pressure, opacity } = opts;
  const strength = (opacity / 100) * 0.2;

  for (let ci = 1; ci < pts.length; ci++) {
    const p = pts[ci];
    const pMod = getEffectivePressure(p, pressure);
    const localR = size * (hasPenInput ? 0.5 + p.p * 0.5 : 1);
    const ri = Math.ceil(localR);
    const r2 = localR * localR;
    const cx = Math.round(p.x), cy = Math.round(p.y);

    for (let dy = -ri; dy <= ri; dy++) {
      for (let dx = -ri; dx <= ri; dx++) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = cx + dx, py = cy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;

        const i = py * W + px;
        const falloff = 1 - Math.sqrt(dd) / localR;
        const hVal = sampleH(px, py);
        const mask = hVal * falloff * falloff * strength * pMod;
        const keep = 1 - clamp(mask, 0, 1);

        wetAlpha[i] *= keep;
        wetness[i] *= keep;
        if (wetAlpha[i] < 1) wetAlpha[i] = 0;
      }
    }
  }
}
```

**2. applyWaterStroke(rawPts, opts) -- stroke-completion tool**

Adds wetness without depositing color. Walks all points in the stroke.

```javascript
function applyWaterStroke(rawPts, opts) {
  if (rawPts.length < 2) return;
  const { size, waterAmount } = opts;
  const amount = waterAmount / 100;

  for (let ci = 0; ci < rawPts.length; ci++) {
    const p = rawPts[ci];
    const pMod = hasPenInput ? 0.3 + p.p * 0.7 : 1;
    const localR = size * (hasPenInput ? 0.5 + p.p * 0.5 : 1);
    const ri = Math.ceil(localR);
    const r2 = localR * localR;
    const cx = Math.round(p.x), cy = Math.round(p.y);

    for (let dy = -ri; dy <= ri; dy++) {
      for (let dx = -ri; dx <= ri; dx++) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = cx + dx, py = cy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;

        const i = py * W + px;
        const hVal = sampleH(px, py);
        if (hVal <= 0) continue;

        wetness[i] = Math.min(1000, wetness[i] + hVal * amount * pMod * 800);
      }
    }
  }
}
```

**3. applySmearChunk(pts, opts) -- real-time tool**

Captures wet paint at previous cursor position, deposits at current position with blending. Uses brush grain for texture variation in the deposit.

```javascript
function applySmearChunk(pts, opts) {
  if (pts.length < 2) return;
  const { size, opacity, pressure } = opts;
  const strength = (opacity / 100) * 0.6;

  for (let ci = 1; ci < pts.length; ci++) {
    const p = pts[ci], pp = pts[ci - 1];
    const pMod = getEffectivePressure(p, pressure);
    const localR = size * (hasPenInput ? 0.5 + p.p * 0.5 : 1);
    const ri = Math.ceil(localR);
    const r2 = localR * localR;

    // Capture paint at previous position (area-weighted sample)
    const prevCx = Math.round(pp.x), prevCy = Math.round(pp.y);
    let sumR = 0, sumG = 0, sumB = 0, sumA = 0, sumW = 0;
    let capCount = 0;
    for (let dy = -ri; dy <= ri; dy++) {
      for (let dx = -ri; dx <= ri; dx++) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = prevCx + dx, py = prevCy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;
        const i = py * W + px;
        if (wetAlpha[i] < 1) continue;
        const f = 1 - Math.sqrt(dd) / localR;
        sumR += wetR[i] * f; sumG += wetG[i] * f; sumB += wetB[i] * f;
        sumA += wetAlpha[i] * f; sumW += f;
        capCount++;
      }
    }
    if (sumW < 0.01) continue;
    const avgR = sumR / sumW, avgG = sumG / sumW, avgB = sumB / sumW;
    const avgA = sumA / sumW;

    // Deposit at current position with brush grain modulation
    const cx = Math.round(p.x), cy = Math.round(p.y);
    for (let dy = -ri; dy <= ri; dy++) {
      for (let dx = -ri; dx <= ri; dx++) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = cx + dx, py = cy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;

        const i = py * W + px;
        const falloff = 1 - Math.sqrt(dd) / localR;
        const hVal = sampleH(px, py);
        const grainVal = sampleBrushGrain(px, py); // brush grain for smear
        const mask = hVal * grainVal * falloff * falloff * strength * pMod;

        // Blend captured paint into existing wet paint
        const blend = clamp(mask, 0, 1);
        wetR[i] = wetR[i] * (1 - blend) + avgR * blend;
        wetG[i] = wetG[i] * (1 - blend) + avgG * blend;
        wetB[i] = wetB[i] * (1 - blend) + avgB * blend;
        wetAlpha[i] = Math.min(200000, wetAlpha[i] * (1 - blend) + avgA * blend);
        wetness[i] = Math.min(1000, wetness[i] + mask * 200);
      }
    }
  }
}
```

**4. applyBlendStroke(rawPts, opts) -- stroke-completion tool**

Two-pass per point: first compute weighted average of wet paint in radius, then blend each pixel toward that average.

```javascript
function applyBlendStroke(rawPts, opts) {
  if (rawPts.length < 2) return;
  const { size, opacity, pressure } = opts;
  const strength = (opacity / 100) * 0.5;

  for (let ci = 0; ci < rawPts.length; ci++) {
    const p = rawPts[ci];
    const pMod = getEffectivePressure(p, pressure);
    const localR = size * (hasPenInput ? 0.5 + p.p * 0.5 : 1);
    const ri = Math.ceil(localR);
    const r2 = localR * localR;
    const cx = Math.round(p.x), cy = Math.round(p.y);

    // Pass 1: compute weighted average of all wet paint in radius
    let sumR = 0, sumG = 0, sumB = 0, sumA = 0, sumW = 0;
    const step = localR > 30 ? 2 : 1; // sparse sampling for large brushes (Open Q #2 from RESEARCH)
    for (let dy = -ri; dy <= ri; dy += step) {
      for (let dx = -ri; dx <= ri; dx += step) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = cx + dx, py = cy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;
        const i = py * W + px;
        if (wetAlpha[i] < 1) continue;
        const f = 1 - Math.sqrt(dd) / localR;
        sumR += wetR[i] * f; sumG += wetG[i] * f; sumB += wetB[i] * f;
        sumA += wetAlpha[i] * f; sumW += f;
      }
    }
    if (sumW < 0.01) continue;
    const avgR = sumR / sumW, avgG = sumG / sumW, avgB = sumB / sumW;
    const avgA = sumA / sumW;

    // Pass 2: blend each pixel toward the average
    for (let dy = -ri; dy <= ri; dy++) {
      for (let dx = -ri; dx <= ri; dx++) {
        const dd = dx * dx + dy * dy;
        if (dd > r2) continue;
        const px = cx + dx, py = cy + dy;
        if (px < 0 || px >= W || py < 0 || py >= H) continue;

        const i = py * W + px;
        if (wetAlpha[i] < 1) continue;
        const falloff = 1 - Math.sqrt(dd) / localR;
        const hVal = sampleH(px, py);
        const mask = hVal * falloff * falloff * strength * pMod;
        const blend = clamp(mask, 0, 1);

        wetR[i] = wetR[i] * (1 - blend) + avgR * blend;
        wetG[i] = wetG[i] * (1 - blend) + avgG * blend;
        wetB[i] = wetB[i] * (1 - blend) + avgB * blend;
        wetAlpha[i] = Math.min(200000, wetAlpha[i] * (1 - blend) + avgA * blend);
      }
    }
  }
}
```

**5. applyBlowChunk(pts, opts) -- real-time tool**

Adds wetness biased in pointer movement direction so flow field carries paint that way.

```javascript
function applyBlowChunk(pts, opts) {
  if (pts.length < 2) return;
  const { size, waterAmount } = opts;
  const strength = (waterAmount || 50) / 100;

  for (let ci = 1; ci < pts.length; ci++) {
    const p = pts[ci], pp = pts[ci - 1];
    let ddx = p.x - pp.x, ddy = p.y - pp.y;
    const dl = Math.hypot(ddx, ddy) || 1;
    ddx /= dl; ddy /= dl;

    const pMod = hasPenInput ? 0.3 + p.p * 0.7 : 1;
    const localR = size * (hasPenInput ? 0.5 + p.p * 0.5 : 1);
    const ri = Math.ceil(localR);
    const r2 = localR * localR;
    const cx = Math.round(p.x), cy = Math.round(p.y);

    for (let oy = -ri; oy <= ri; oy++) {
      for (let ox = -ri; ox <= ri; ox++) {
        const dd = ox * ox + oy * oy;
        if (dd > r2) continue;
        const px = cx + ox, py = cy + oy;
        if (px < 1 || px >= W - 1 || py < 1 || py >= H - 1) continue;

        const i = py * W + px;
        if (wetness[i] < 1 && wetAlpha[i] < 1) continue;

        const falloff = 1 - Math.sqrt(dd) / localR;
        const hVal = sampleH(px, py);
        const mask = hVal * falloff * falloff * falloff * strength * pMod;

        wetness[i] = Math.min(1000, wetness[i] + mask * 500);
      }
    }
  }
}
```
