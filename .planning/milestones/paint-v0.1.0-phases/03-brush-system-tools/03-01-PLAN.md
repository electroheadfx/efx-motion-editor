**5. Implement brush grain quadrant mirroring at load time (per D-08, D-09).**

After the existing `brushGrain` loading code (around line ~204), add:

```javascript
let mirroredGrain = null;
const MIRRORED_GRAIN_SIZE = 256;

function createMirroredBrushGrain(src128) {
  const S = 128;
  const mirrored = new Float32Array(S * 2 * S * 2);
  for (let y = 0; y < S; y++) {
    for (let x = 0; x < S; x++) {
      const val = src128[y * S + x];
      mirrored[y * S * 2 + x] = val;
      mirrored[y * S * 2 + (S * 2 - 1 - x)] = val;
      mirrored[(S * 2 - 1 - y) * S * 2 + x] = val;
      mirrored[(S * 2 - 1 - y) * S * 2 + (S * 2 - 1 - x)] = val;
    }
  }
  return mirrored;
}

function sampleBrushGrain(x, y) {
  if (!mirroredGrain) return 1.0;
  const S = MIRRORED_GRAIN_SIZE;
  const ix = ((Math.floor(x) % S) + S) % S;
  const iy = ((Math.floor(y) % S) + S) % S;
  return mirroredGrain[iy * S + ix];
}
```

Then modify the brushGrain loading callback to also create the mirrored version:
```javascript

```
