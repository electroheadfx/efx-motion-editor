//  NOISE — deterministic hash-based (from research/watercolor-canvas.html)
//  Required for deterministic replay (D-06): no Math.random() in noise path
// ============================================================
function _dot(x1, y1, x2, y2) { return x1 * x2 + y1 * y2; }
function _fract(x) { return x - Math.floor(x); }
function _rand(x, y) {
  return _fract(Math.sin(_dot(x, y, 12.9898, 4.1414)) * 43758.5453);
}
function noise(px, py) {
  const ix = Math.floor(px), iy = Math.floor(py);
  let ux = px - ix, uy = py - iy;
  ux = ux * ux * (3.0 - 2.0 * ux);
  uy = uy * uy * (3.0 - 2.0 * uy);
  const a = _rand(ix, iy), b = _rand(ix + 1, iy);
  const c = _rand(ix, iy + 1), d = _rand(ix + 1, iy + 1);
  const res = a + (b - a) * ux + (c - a) * uy + (a - b - c + d) * ux * uy;
  return res * res;
}
function fbm(x, y, octaves) {
  let v = 0.0, a = 0.5;
  let shiftX = 100, shiftY = 100;
  const cosR = Math.cos(0.5), sinR = Math.sin(0.5);
  for (let i = 0; i < octaves; i++) {
    v += a * noise(x, y);
    const nx = cosR * x + sinR * y;
    const ny = -sinR * x + cosR * y;
    x = nx * 2.0 + shiftX;
    y = ny * 2.0 + shiftY;
    a *= 0.5;
  }
  return v;
}

// Bilinear interpolation for sub-pixel displaced reads (from research prototype)
function sampleChannel(buf, fx, fy) {
  fx = Math.max(0, Math.min(W - 1.001, fx));
  fy = Math.max(0, Math.min(H - 1.001, fy));
  const x0 = Math.floor(fx), y0 = Math.floor(fy);
  const x1 = Math.min(x0 + 1, W - 1);
  const y1 = Math.min(y0 + 1, H - 1);
  const tx = fx - x0, ty = fy - y0;
  const a = buf[y0 * W + x0], b = buf[y0 * W + x1];
  const c = buf[y1 * W + x0], d = buf[y1 * W + x1];
  return (a * (1 - tx) + b * tx) * (1 - ty) +
         (c * (1 - tx) + d * tx) * ty;
}
```

// ============================================================
//  DISPLACEMENT MAP — FBM noise + paper grain heightmap (D-03)
//  Computed once at init, recomputed when paper texture changes.
//  Paper gradient adds directional displacement (valleys catch paint).
//  Weight: 0.7 FBM + 0.3 paper gradient (per research recommendation).
// ============================================================
const FBM_SCALE = 22.0;
const FBM_OCTAVES = 4;
const DISP_STRENGTH = 0.005; // in UV space; at 1000px width -> ~5px displacement
const PAPER_INFLUENCE = 0.3;

function precomputeDisplacement() {
  const aspect = H / W;
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      const u = x / W, v = y / H;
      const fbmVal = fbm(u * FBM_SCALE, v * FBM_SCALE, FBM_OCTAVES);

      let paperGradX = 0, paperGradY = 0;
      if (paperHeight) {
        const idx = y * W + x;
        const hL = x > 0 ? paperHeight[idx - 1] : paperHeight[idx];
        const hR = x < W - 1 ? paperHeight[idx + 1] : paperHeight[idx];
        const hU = y > 0 ? paperHeight[idx - W] : paperHeight[idx];
        const hD = y < H - 1 ? paperHeight[idx + W] : paperHeight[idx];
        paperGradX = (hR - hL) * PAPER_INFLUENCE;
        paperGradY = (hD - hU) * PAPER_INFLUENCE;
      }

      const idx = y * W + x;
      dispPxX[idx] = (fbmVal * DISP_STRENGTH + paperGradX * DISP_STRENGTH) * W;
      dispPxY[idx] = (fbmVal * DISP_STRENGTH + paperGradY * DISP_STRENGTH) * H * aspect;
    }
  }
}
```

**Key domain adaptation (Pitfall 1 from research):**
The research prototype uses greyscale where 1.0 = no paint (white) and 0.0 = full paint (black), with `min()` for darkening. Our codebase uses:
- `wetAlpha`: 0-200000 where HIGHER = MORE paint density
- `wetR/G/B`: 0-255 color channels

Therefore the darken flood must use `max()` on wetAlpha (higher density = more paint), and the self-stabilization divides instead of multiplies (to slightly reduce the flood value). Color channels follow alpha flow via weighted blending.

```javascript
// ============================================================
//  DIFFUSE STEP — FBM-displaced darken flood (per D-01, D-02)
//  Replaces flowStep(). Adapted from research/watercolor-canvas.html diffuse().
//
//  Algorithm:
//  1. For each wet pixel, sample 4 displaced neighbors (using precomputed FBM displacement)
//  2. Flood: take max density of displaced neighbors (max because higher wetAlpha = more paint)
//  3. Stabilize: divide flood by 1.02 to prevent infinite spreading
//  4. Blend flood result with original value (30/70 split)
//  5. Color follows alpha: RGB blended toward highest-alpha displaced neighbor
//  6. Gentle drying: slight alpha drain per pass
// ============================================================
function diffuseStep() {
  const w = W, h = H;

  // Phase 1: Diffuse wetAlpha (density) and track which displaced neighbor contributed most
  for (let y = 0; y < h; y++) {
    const yw = y * w;
    for (let x = 0; x < w; x++) {
      const idx = yw + x;
      const prev = wetAlpha[idx];

      if (prev < DIFF_THRESHOLD) {
        tmpAlpha[idx] = prev;
        tmpR[idx] = wetR[idx];
        tmpG[idx] = wetG[idx];
        tmpB[idx] = wetB[idx];
        continue;
      }

      const dx = dispPxX[idx], dy = dispPxY[idx];

      // Sample center + 4 displaced neighbors for alpha (density)
      const s1 = sampleChannel(wetAlpha, x + dx, y);
      const s2 = sampleChannel(wetAlpha, x - dx, y);
      const s3 = sampleChannel(wetAlpha, x, y + dy);
      const s4 = sampleChannel(wetAlpha, x, y - dy);

      // Flood: maximum density of displaced neighbors (higher wetAlpha = more paint)
      let flood = Math.max(prev, s1, s2, s3, s4);

      // Self-stabilization: slightly reduce flood to prevent infinite spreading
      flood = flood / DIFF_STABILIZE;

      // Blend: 30% flood + 70% original
      const brightened = Math.max(prev, flood);
      let newAlpha = brightened * DIFF_BLEND_RATE + prev * (1 - DIFF_BLEND_RATE);

      // Gentle drying: slight alpha drain per pass
      newAlpha = Math.max(0, newAlpha - DIFF_GENTLE_DRY);

      tmpAlpha[idx] = newAlpha;

      // Phase 2: Color follows alpha flow
      // Sample displaced neighbor colors and blend toward the highest-alpha neighbor
      const prevR = wetR[idx], prevG = wetG[idx], prevB = wetB[idx];

      // Sample colors at same displaced positions
      const r1 = sampleChannel(wetR, x + dx, y), g1 = sampleChannel(wetG, x + dx, y), b1 = sampleChannel(wetB, x + dx, y);
      const r2 = sampleChannel(wetR, x - dx, y), g2 = sampleChannel(wetG, x - dx, y), b2 = sampleChannel(wetB, x - dx, y);
      const r3 = sampleChannel(wetR, x, y + dy), g3 = sampleChannel(wetG, x, y + dy), b3 = sampleChannel(wetB, x, y + dy);
      const r4 = sampleChannel(wetR, x, y - dy), g4 = sampleChannel(wetG, x, y - dy), b4 = sampleChannel(wetB, x, y - dy);

      // Alpha-weighted color average of displaced neighbors
      const totalA = s1 + s2 + s3 + s4;
      if (totalA > 0) {
        const avgR = (r1 * s1 + r2 * s2 + r3 * s3 + r4 * s4) / totalA;
        const avgG = (g1 * s1 + g2 * s2 + g3 * s3 + g4 * s4) / totalA;
        const avgB = (b1 * s1 + b2 * s2 + b3 * s3 + b4 * s4) / totalA;

        // Blend color: 30% neighbor average, 70% original (same rate as alpha)
        tmpR[idx] = prevR * (1 - DIFF_BLEND_RATE) + avgR * DIFF_BLEND_RATE;
        tmpG[idx] = prevG * (1 - DIFF_BLEND_RATE) + avgG * DIFF_BLEND_RATE;
        tmpB[idx] = prevB * (1 - DIFF_BLEND_RATE) + avgB * DIFF_BLEND_RATE;
      } else {
        tmpR[idx] = prevR;
        tmpG[idx] = prevG;
        tmpB[idx] = prevB;
      }
    }
  }

  // Phase 3: Swap — copy temp buffers back to wet arrays (ping-pong complete)
  wetAlpha.set(tmpAlpha);
  wetR.set(tmpR);
  wetG.set(tmpG);
  wetB.set(tmpB);
}
```
