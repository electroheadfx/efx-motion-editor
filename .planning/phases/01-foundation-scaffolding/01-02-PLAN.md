    A2. Create src/scenes/testScene.tsx -- Test scene with JSX pragma:
    ```tsx
    /** @jsxImportSource @efxlab/motion-canvas-2d/lib */
    import { makeScene2D, Img } from '@efxlab/motion-canvas-2d';
    import { createRef, waitFor } from '@efxlab/motion-canvas-core';
    import testImage from '../assets/test-image.jpg';

    export default makeScene2D(function* (view) {
      const imageRef = createRef<Img>();
      view.add(
        <Img
          ref={imageRef}
          src={testImage}
          width={1920}
          height={1080}
        />
      );
      yield* waitFor(1);
    });
    ```

    **CRITICAL:** The `/** @jsxImportSource @efxlab/motion-canvas-2d/lib */` pragma MUST be the very first line. This overrides the global Preact JSX for this file only. Without it, the scene JSX will use Preact's h() function instead of Motion Canvas nodes, and rendering will break silently.

    **CRITICAL:** The `?scene` suffix on the import in project.ts is required -- the Motion Canvas vite plugin intercepts this query parameter to apply scene transforms.

    A3. Create src/components/Preview.tsx -- Preact wrapper for the player:
    ```tsx
    import { useRef } from 'preact/hooks';
    import '@efxlab/motion-canvas-player';

    export function Preview() {
      const playerRef = useRef<HTMLElement>(null);

      return (
        <div class="relative w-full aspect-video bg-black rounded overflow-hidden">
          <motion-canvas-player
            ref={playerRef}
            src="/src/project.ts?project"
            auto
            responsive
            background="#000000"
          />
        </div>
      );
    }
    ```

    Note: The `src` attribute uses `/src/project.ts?project` -- the `?project` query parameter triggers the vite plugin to serve the bootstrapped project module. The `auto` attribute starts playback automatically. The `responsive` attribute makes the player fill its container.

    **If the player shows a loading error:** Check the browser console for the dynamic import path. The path must match what the vite dev server can resolve. Try variations: `"/src/project.ts?project"`, `"./project.ts?project"`, or an absolute URL.

    **Part B: Signal Stores**

    Create all 6 stores following the locked decisions:
    - One signal per field for fine-grained reactivity
    - Store objects with methods pattern (e.g., projectStore.setName())
    - Stores can import from other stores directly
    - History store: skeleton only (HistoryEntry type + basic signals, no undo/redo logic)

    B1. src/stores/projectStore.ts:
    ```typescript
    import { signal, computed } from '@preact/signals';
    import type { ProjectData } from '../types/project';

    const name = signal('Untitled Project');
    const fps = signal(24);
    const width = signal(1920);
    const height = signal(1080);

    const aspectRatio = computed(() => width.value / height.value);

    export const projectStore = {
      name, fps, width, height, aspectRatio,

      setName(v: string) { name.value = v; },
      setFps(v: number) { fps.value = v; },
      setResolution(w: number, h: number) { width.value = w; height.value = h; },

      loadFromData(data: ProjectData) {
        name.value = data.name;
        fps.value = data.fps;
        width.value = data.width;
        height.value = data.height;
      },

      reset() {
        name.value = 'Untitled Project';
        fps.value = 24;
        width.value = 1920;
        height.value = 1080;
      },
    };
    ```

    B2. src/stores/sequenceStore.ts:
    ```typescript
    import { signal } from '@preact/signals';
    import type { Sequence } from '../types/sequence';

    const sequences = signal<Sequence[]>([]);
    const activeSequenceId = signal<string | null>(null);

    export const sequenceStore = {
      sequences, activeSequenceId,

      add(seq: Sequence) { sequences.value = [...sequences.value, seq]; },
      remove(id: string) { sequences.value = sequences.value.filter(s => s.id !== id); },
      setActive(id: string | null) { activeSequenceId.value = id; },
      getById(id: string) { return sequences.value.find(s => s.id === id) ?? null; },
      reset() { sequences.value = []; activeSequenceId.value = null; },
    };
    ```

    B3. src/stores/layerStore.ts:
    ```typescript
    import { signal } from '@preact/signals';
    import type { Layer } from '../types/layer';

    const layers = signal<Layer[]>([]);
    const selectedLayerId = signal<string | null>(null);

    export const layerStore = {
      layers, selectedLayerId,

      add(layer: Layer) { layers.value = [...layers.value, layer]; },
      remove(id: string) { layers.value = layers.value.filter(l => l.id !== id); },
      setSelected(id: string | null) { selectedLayerId.value = id; },
      updateLayer(id: string, updates: Partial<Layer>) {
        layers.value = layers.value.map(l => l.id === id ? { ...l, ...updates } : l);
      },
      reorder(fromIndex: number, toIndex: number) {
        const arr = [...layers.value];
        const [moved] = arr.splice(fromIndex, 1);
        arr.splice(toIndex, 0, moved);
        layers.value = arr;
      },
      reset() { layers.value = []; selectedLayerId.value = null; },
    };
    ```

    B4. src/stores/timelineStore.ts:
    ```typescript
    import { signal, computed } from '@preact/signals';
    import { projectStore } from './projectStore';

    const currentFrame = signal(0);
    const isPlaying = signal(false);
    const zoom = signal(1);
    const scrollX = signal(0);

    const currentTime = computed(() => currentFrame.value / projectStore.fps.value);

    export const timelineStore = {
      currentFrame, isPlaying, zoom, scrollX, currentTime,

      seek(frame: number) { currentFrame.value = Math.max(0, frame); },
      setPlaying(v: boolean) { isPlaying.value = v; },
      togglePlaying() { isPlaying.value = !isPlaying.value; },
      stepForward() { currentFrame.value += 1; },
      stepBackward() { currentFrame.value = Math.max(0, currentFrame.value - 1); },
      setZoom(v: number) { zoom.value = Math.max(0.1, Math.min(10, v)); },
      setScrollX(v: number) { scrollX.value = v; },
      reset() { currentFrame.value = 0; isPlaying.value = false; zoom.value = 1; scrollX.value = 0; },
    };
    ```
