    const LUMA_WEIGHTS = { r: 0.2126, g: 0.7152, b: 0.0722 }; // ITU-R BT.709

    /**
     * Apply luma key to an offscreen canvas.
     * @param canvas - HTMLCanvasElement with paint strokes on white background
     * @param invert - if true, apply luma invert (black strokes → white opaque, transparent BG)
     *                  if false, apply luma key (white BG → transparent, strokes opaque)
     * Uses ITU-R BT.709 luma weights.
     * IMPORTANT: This modifies the canvas ImageData in-place. Call on a copy if you need
     * to preserve the original. Never call on _frameFxCache directly (pitfall 3).
     */
    export function applyLumaKey(canvas: HTMLCanvasElement, invert: boolean): void {
      const ctx = canvas.getContext('2d');
      if (!ctx) return;
      const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
      const data = imgData.data;
      for (let i = 0; i < data.length; i += 4) {
        const r = data[i];
        const g = data[i + 1];
        const b = data[i + 2];
        const luma = LUMA_WEIGHTS.r * r + LUMA_WEIGHTS.g * g + LUMA_WEIGHTS.b * b;
        if (invert) {
          data[i + 3] = luma;         // Luma Invert: white BG → transparent, black strokes → opaque
        } else {
          data[i + 3] = 255 - luma;  // Luma Key: white BG → transparent, colored strokes → opaque
        }
      }
      ctx.putImageData(imgData, 0, 0);
    }
