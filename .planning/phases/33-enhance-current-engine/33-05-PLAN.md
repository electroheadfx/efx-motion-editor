```typescript
/** RGB to HSV (for Box mode) */
export function rgbToHsv(r: number, g: number, b: number): { h: number; s: number; v: number } {
  r /= 255; g /= 255; b /= 255;
  const max = Math.max(r, g, b), min = Math.min(r, g, b);
  const d = max - min;
  const v = max;
  const s = max === 0 ? 0 : d / max;
  let h = 0;
  if (d !== 0) {
    if (max === r) h = ((g - b) / d + (g < b ? 6 : 0)) / 6;
    else if (max === g) h = ((b - r) / d + 2) / 6;
    else h = ((r - g) / d + 4) / 6;
  }
  return { h: h * 360, s, v };
}

/** HSV to RGB */
export function hsvToRgb(h: number, s: number, v: number): { r: number; g: number; b: number } {
  h /= 360;
  const i = Math.floor(h * 6);
  const f = h * 6 - i;
  const p = v * (1 - s);
  const q = v * (1 - f * s);
  const t = v * (1 - (1 - f) * s);
  let r = 0, g = 0, b = 0;
  switch (i % 6) {
    case 0: r = v; g = t; b = p; break;
    case 1: r = q; g = v; b = p; break;
    case 2: r = p; g = v; b = t; break;
    case 3: r = p; g = q; b = v; break;
    case 4: r = t; g = p; b = v; break;
    case 5: r = v; g = p; b = q; break;
  }
  return { r: Math.round(r * 255), g: Math.round(g * 255), b: Math.round(b * 255) };
}

/** RGB to CMYK (simple mathematical, no ICC profiles) */
export function rgbToCmyk(r: number, g: number, b: number): { c: number; m: number; y: number; k: number } {
  r /= 255; g /= 255; b /= 255;
  const k = 1 - Math.max(r, g, b);
  if (k === 1) return { c: 0, m: 0, y: 0, k: 1 };
  return {
    c: (1 - r - k) / (1 - k),
    m: (1 - g - k) / (1 - k),
    y: (1 - b - k) / (1 - k),
    k,
  };
}

/** CMYK to RGB */
export function cmykToRgb(c: number, m: number, y: number, k: number): { r: number; g: number; b: number } {
  return {
    r: Math.round(255 * (1 - c) * (1 - k)),
    g: Math.round(255 * (1 - m) * (1 - k)),
    b: Math.round(255 * (1 - y) * (1 - k)),
  };
}
```

**2. Create `app/src/components/sidebar/InlineColorPicker.tsx`:**

The component has this structure:
- **Header:** 4 mode tabs (Box, TSL, RVB, CMYK) + close button (X icon)
- **Mode content area** (switches based on active tab):
  - **Box (default):** HSV color square (256x256 canvas) where X=saturation (0-1), Y=value (1-0 top to bottom). Below: Hue slider (0-360), Alpha slider (0-1).
  - **TSL:** Four labeled sliders: H (0-360), S (0-100%), L (0-100%), A (0-100%)
  - **RVB:** Four labeled sliders: R (0-255), G (0-255), B (0-255), A (0-100%)
  - **CMYK:** Five labeled sliders: C (0-100%), M (0-100%), Y (0-100%), K (0-100%), A (0-100%)
- **HEX input field:** Text input showing current hex value (e.g., `#203769`), editable. Validate on change, apply if valid 3/4/6/8 digit hex.
- **Swatches section:**
  - Row labeled "Recent" — auto-collected colors (up to 16 small squares)
  - Row labeled "Favorites" — user-managed colors. Click to apply. Right-click or long-press to remove. "+" button at end to add current color to favorites.

**Props interface:**
```typescript
interface InlineColorPickerProps {
  color: string;          // current hex color
  opacity: number;        // current alpha 0-1
  onChange: (color: string, opacity: number) => void;  // fires on every interaction
  onClose: () => void;
}
```

**Key implementation details:**
- HSV square in Box mode: render via canvas element. On pointer down/move, sample (x, y) to get (saturation, value), combine with current hue to produce RGB, call onChange.
- All sliders fire onChange on input event (realtime, not on release).
- Collect recent color on pointer up in HSV square / on slider change end (debounced). Do NOT collect intermediate slider values.
- Component uses `position: fixed` or `absolute`, docked to the left side of the canvas area. Z-index above canvas but below modals.
- Width: ~260px. Dark background matching app theme.
