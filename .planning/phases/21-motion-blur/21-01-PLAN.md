       ```typescript
       import type {KeyframeValues} from '../types/layer';

       export interface LayerVelocity {
         dx: number;       // pixels/frame
         dy: number;       // pixels/frame
         dRotation: number; // degrees/frame
         dScale: number;    // factor/frame
       }

       const VELOCITY_THRESHOLD = 0.5;

       export function computeLayerVelocity(
         current: KeyframeValues,
         previous: KeyframeValues,
       ): LayerVelocity {
         return {
           dx: current.x - previous.x,
           dy: current.y - previous.y,
           dRotation: current.rotation - previous.rotation,
           dScale: (current.scaleX - previous.scaleX + current.scaleY - previous.scaleY) / 2,
         };
       }

       export function isStationary(v: LayerVelocity): boolean {
         return Math.abs(v.dx) + Math.abs(v.dy) + Math.abs(v.dRotation) + Math.abs(v.dScale) < VELOCITY_THRESHOLD;
       }
       ```

    Create `Application/src/lib/glMotionBlur.ts` as a structural clone of `glBlur.ts` with a different shader and different uniforms. Follow the identical lazy-init WebGL2 singleton pattern.

    **Vertex shader** (same as glBlur.ts):
    ```glsl
    #version 300 es
    layout(location = 0) in vec2 a_position;
    out vec2 vUV;
    void main() {
      gl_Position = vec4(a_position, 0.0, 1.0);
      vUV = a_position * 0.5 + 0.5;
    }
    ```

    **Fragment shader** (directional motion blur from spec):
    ```glsl
    #version 300 es
    precision highp float;

    uniform sampler2D iChannel0;
    uniform vec2 iResolution;
    uniform vec2 uVelocity;    // velocity in pixels (dx, dy)
    uniform float uStrength;   // blur strength (0.0 - 1.0) = shutterAngle / 360
    uniform int uSamples;      // 4, 8, or 16

    in vec2 vUV;
    out vec4 fragColor;

    void main() {
        vec2 texelSize = 1.0 / iResolution;
        vec2 velocity = uVelocity * texelSize * uStrength;

        vec4 color = vec4(0.0);
        float totalWeight = 0.0;

        for (int i = 0; i < uSamples; i++) {
            float t = float(i) / float(uSamples - 1) - 0.5;
            vec2 offset = velocity * t;
            vec2 sampleUV = clamp(vUV + offset, vec2(0.0), vec2(1.0));
            float weight = 1.0 - abs(t * 2.0); // triangle filter
            color += texture(iChannel0, sampleUV) * weight;
            totalWeight += weight;
        }

        fragColor = color / totalWeight;
    }
    ```

    **Module structure:**
    1. `interface MotionBlurResources` with fields: `program`, `texSource`, `fbo`, `texFBO`, `vao`, `uIChannel0`, `uIResolution`, `uVelocity`, `uStrength`, `uSamples`, `width`, `height`
    2. Singleton `_gl`, `_glCanvas`, `_resources`, `_initFailed` (identical to glBlur.ts pattern)
    3. `getGL()` — create canvas, get WebGL2 context with `{ premultipliedAlpha: false, preserveDrawingBuffer: true, antialias: false, depth: false, stencil: false }`, handle context lost
    4. `compileShader(gl, type, source)` — compile + error check (copy from glBlur.ts)
    5. `createResources(gl, width, height)` — compile shaders, link program, create fullscreen quad VAO (Float32Array([-1,-1,1,-1,-1,1,1,1]) TRIANGLE_STRIP), create textures with `texStorage2D(TEXTURE_2D, 1, RGBA8, w, h)`, create FBO, get uniform locations
    6. `ensureResources(gl, w, h)` — recreate if size changed (delete old, create new)
    7. `export function applyMotionBlur(source: HTMLCanvasElement, targetCtx: CanvasRenderingContext2D, velocity: {dx: number; dy: number}, strength: number, samples: number, width: number, height: number): boolean`
       - Call `getGL()`, return false if null
       - Call `ensureResources(gl, width, height)`, return false if null
       - Resize `_glCanvas` if needed
       - Upload source: `gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true)` then `gl.texSubImage2D(gl.TEXTURE_2D, 0, 0, 0, gl.RGBA, gl.UNSIGNED_BYTE, source)` then restore `UNPACK_FLIP_Y_WEBGL` to false
       - Bind FBO, set viewport, use program
       - Set uniforms: `uniform2f(uIResolution, w, h)`, `uniform2f(uVelocity, velocity.dx, velocity.dy)`, `uniform1f(uStrength, strength)`, `uniform1i(uSamples, samples)`
       - Bind source texture, bind VAO, draw TRIANGLE_STRIP 4 vertices
       - Read result: unbind FBO, set viewport, clear, draw textured quad from FBO texture to screen, then `targetCtx.drawImage(_glCanvas, 0, 0, width, height)`
       - Return true

    - Set `gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true)` before texSubImage2D, restore to false after (Pitfall 1 from research)
    - Use `texStorage2D` for immutable texture allocation + `texSubImage2D` for updates (not texImage2D every frame)
    - Handle `uSamples` as `uniform1i` (integer uniform, not float)
