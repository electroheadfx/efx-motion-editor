```typescript
export function AddKeyPhotoButton({sequenceId}: {sequenceId: string}) {
  const [popoverOpen, setPopoverOpen] = useState(false);
  const popRef = useRef<HTMLDivElement>(null);
  const importedImages = imageStore.images.value;

  // Close popover on click outside
  useEffect(() => {
    if (!popoverOpen) return;
    function handleClick(e: MouseEvent) {
      if (popRef.current && !popRef.current.contains(e.target as Node)) {
        setPopoverOpen(false);
      }
    }
    document.addEventListener('mousedown', handleClick);
    return () => document.removeEventListener('mousedown', handleClick);
  }, [popoverOpen]);

  return (
    <div class="relative shrink-0">
      <button onClick={() => setPopoverOpen(!popoverOpen)}>+</button>
      {popoverOpen && (
        <div ref={popRef} class="absolute right-0 top-7 z-50 bg-[#1E1E1E] border border-[#333] rounded-md shadow-xl p-2 min-w-[180px] max-w-[260px] max-h-[300px] overflow-y-auto">
          {importedImages.length === 0 ? (
            <span>No imported images. Import images first.</span>
          ) : (
            <div class="grid grid-cols-4 gap-1">
              {importedImages.map((img) => (
                <button
                  key={img.id}
                  style={{ backgroundImage: `url(${assetUrl(img.thumbnail_path)})` }}
                  onClick={() => handleAddImage(img.id)}
                />
              ))}
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```
