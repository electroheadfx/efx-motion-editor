  16. Wrap steps 4-15 in try/catch. On error: console.error, set isDecoding=false.

```typescript
export const audioTrackLayouts = computed<AudioTrackLayout[]>(() => {
  const tracks = audioStore.tracks.value;
  const selectedId = audioStore.selectedTrackId.value;
  return tracks
    .slice()
    .sort((a, b) => a.order - b.order)
    .map(track => ({
      trackId: track.id,
      trackName: track.name,
      offsetFrame: track.offsetFrame,
      inFrame: track.inFrame,
      outFrame: track.outFrame,
      muted: track.muted,
      volume: track.volume,
      peaks: audioPeaksCache.get(track.id) ?? { tier1: new Float32Array(0), tier2: new Float32Array(0), tier3: new Float32Array(0) },
      trackHeight: track.trackHeight,
      fadeInFrames: track.fadeInFrames,
      fadeOutFrames: track.fadeOutFrames,
      fadeInCurve: track.fadeInCurve,
      fadeOutCurve: track.fadeOutCurve,
      slipOffset: track.slipOffset,
      selected: track.id === selectedId,
    }));
});
```

Add private method `drawAudioTrack(ctx, track: AudioTrackLayout, y: number, frameWidth: number, scrollX: number, canvasWidth: number)`:
1. Draw track background: `ctx.fillStyle = colors.audioTrackBg; ctx.fillRect(0, y, canvasWidth, track.trackHeight)`
2. Draw header background (0 to TRACK_HEADER_WIDTH): `ctx.fillStyle = colors.audioHeaderBg; ctx.fillRect(0, y, TRACK_HEADER_WIDTH, track.trackHeight)`
3. Draw track name in header (10px font, colors.trackName, truncated to fit TRACK_HEADER_WIDTH - 8px padding):
   ```
   ctx.font = '10px Inter, sans-serif'; ctx.fillStyle = track.muted ? colors.trackName + '66' : colors.trackName;
   ctx.fillText(truncated_name, 8, y + track.trackHeight / 2 + 3);
   ```
4. Draw selected indicator: if `track.selected`, draw 2px left accent border: `ctx.fillStyle = colors.accent (--color-accent); ctx.fillRect(0, y, 2, track.trackHeight)`
5. Compute range bar position:
   ```
   const barX = (track.offsetFrame + track.inFrame) * frameWidth - scrollX + TRACK_HEADER_WIDTH;
   const totalFrames = track.outFrame - track.inFrame;
   const barW = totalFrames * frameWidth;
   const padding = 4; // Range bar vertical padding
   const barY = y + padding;
   const barH = track.trackHeight - padding * 2;
   ```
6. Select resolution tier based on zoom (per D-04):
   ```
   const pixelsPerPeak = barW / (track.peaks.tier2.length / 2);
   const peaks = pixelsPerPeak < 1 ? track.peaks.tier1
     : pixelsPerPeak > 4 ? track.peaks.tier3
     : track.peaks.tier2;
   ```
7. If peaks.length === 0, skip waveform drawing (no audio decoded yet).
8. Draw center line per D-05: `ctx.strokeStyle = colors.audioCenterline; ctx.lineWidth = 0.5;` -- horizontal line at `centerY = barY + barH / 2` from max(barX, TRACK_HEADER_WIDTH) to min(barX + barW, canvasWidth).
9. Draw waveform peaks:
   ```
   ctx.fillStyle = track.muted ? colors.audioWaveformMuted : colors.audioWaveform;
   const peakCount = peaks.length / 2;
   const centerY = barY + barH / 2;
   const halfH = barH / 2 - 1;
   for (let i = 0; i < peakCount; i++) {
     const px = barX + (i / peakCount) * barW;
     if (px + barW / peakCount < TRACK_HEADER_WIDTH || px > canvasWidth) continue; // clip to visible
     const min = peaks[i * 2];
     const max = peaks[i * 2 + 1];
     const top = centerY - max * halfH;
     const bottom = centerY - min * halfH;
     const w = Math.max(1, barW / peakCount);
     ctx.fillRect(px, top, w, bottom - top);
   }
   ```
10. Draw fade overlays per D-20:
    - Fade-in (if track.fadeInFrames > 0): Linear gradient from audioTrackBg (full opacity) at barX to transparent at barX + fadeInFrames * frameWidth. Draw rect over that region.
    - Fade-out (if track.fadeOutFrames > 0): Linear gradient from transparent at barX + barW - fadeOutFrames * frameWidth to audioTrackBg (full opacity) at barX + barW.
11. Draw edge lines at barX and barX+barW: 2px vertical lines in audioWaveform color at full opacity.

Update `computeCanvasHeight()` (or equivalent method that computes total canvas height) to include audio track heights: sum of all audio track trackHeight values.
