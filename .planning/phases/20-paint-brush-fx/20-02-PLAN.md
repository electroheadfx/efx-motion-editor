2. **STAMP_VERT_SRC** -- Vertex shader for stamp quad rendering. Takes a_position (quad corners -1..1), a_texCoord, and uniforms for stamp center (u_center), stamp size (u_size), and canvas resolution (u_resolution). Outputs gl_Position in clip space and v_texCoord for the stamp texture:
```glsl
#version 300 es
layout(location = 0) in vec2 a_position;
layout(location = 1) in vec2 a_texCoord;
uniform vec2 u_center;   // stamp center in pixel coords
uniform vec2 u_size;     // stamp width, height in pixels
uniform vec2 u_resolution; // canvas width, height
out vec2 v_texCoord;
void main() {
  vec2 pixel = u_center + a_position * u_size * 0.5;
  vec2 ndc = (pixel / u_resolution) * 2.0 - 1.0;
  ndc.y = -ndc.y; // flip Y for Canvas 2D compatibility
  gl_Position = vec4(ndc, 0.0, 1.0);
  v_texCoord = a_texCoord;
}
```

3. **STAMP_FRAG_SRC** -- Fragment shader for individual brush stamp rendering. Takes u_color (vec4 RGBA), v_texCoord, and renders a soft circular falloff:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform vec4 u_color;
uniform float u_hardness; // 0 = soft, 1 = hard edge
out vec4 out_fragColor;
void main() {
  float dist = length(v_texCoord - 0.5) * 2.0;
  float alpha = 1.0 - smoothstep(u_hardness, 1.0, dist);
  out_fragColor = vec4(u_color.rgb, u_color.a * alpha);
}
```

4. **SPECTRAL_GLSL** -- The full spectral.js GLSL port (~520 lines). Fetch the complete source from https://github.com/rvanwijnen/spectral.js/blob/master/shader/spectral.glsl and embed as a string constant. This includes:

   - `spectral_mix(vec3 color1, vec3 color2, float t)` function
   - sRGB/linear/XYZ conversion functions
   - All prefixed with `spectral_` namespace

   The source MUST be fetched from the canonical repository. It is MIT-licensed. Do NOT hand-write spectral mixing math.

5. **SPECTRAL_COMPOSITE_FRAG_SRC** -- Fragment shader that composites a new stroke onto the accumulation buffer using spectral mixing. Uses the SPECTRAL_GLSL code (concatenated at compile time). Uniforms: u_existing (sampler2D, accumulated buffer), u_newStroke (sampler2D, new stroke), u_resolution (vec2). The shader reads both textures, calls spectral_mix() based on alpha ratio, and outputs the mixed color:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform sampler2D u_existing;
uniform sampler2D u_newStroke;
out vec4 out_fragColor;

// spectral.glsl functions are prepended at compile time

void main() {
  vec4 existing = texture(u_existing, v_texCoord);
  vec4 incoming = texture(u_newStroke, v_texCoord);
  if (incoming.a < 0.001) {
    out_fragColor = existing;
    return;
  }
  if (existing.a < 0.001) {
    out_fragColor = incoming;
    return;
  }
  float mixFactor = incoming.a / (existing.a + incoming.a + 0.001);
  vec3 mixed = spectral_mix(existing.rgb, incoming.rgb, mixFactor);
  float combinedAlpha = min(1.0, existing.a + incoming.a);
  out_fragColor = vec4(mixed, combinedAlpha);
}
```

Note: When compiling this shader, brushFxRenderer will concatenate: `#version 300 es\nprecision highp float;\n` + SPECTRAL_GLSL + the rest of the fragment source (without the version/precision lines, since they come from the prefix). Export a helper function `buildSpectralCompositeSrc()` that handles this concatenation:
```typescript
export function buildSpectralCompositeSrc(): string {
  return '#version 300 es\nprecision highp float;\n'
    + SPECTRAL_GLSL + '\n'
    + SPECTRAL_COMPOSITE_FRAG_BODY;
}
```
Where SPECTRAL_COMPOSITE_FRAG_BODY is the main() function and its in/uniform/out declarations without version/precision.

2. **GRAIN_POST_FRAG_SRC** -- Fragment shader post-pass for paper grain texture (per D-13, PAINT-10). Reads the accumulated paint texture, applies multi-octave simplex noise to simulate paper absorption:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform sampler2D u_input;
uniform float u_grain;       // 0-1 grain intensity
uniform vec2 u_resolution;
out vec4 out_fragColor;

// simplex noise functions prepended at compile time

void main() {
  vec4 color = texture(u_input, v_texCoord);
  if (color.a < 0.001) { out_fragColor = color; return; }
  vec2 noiseCoord = gl_FragCoord.xy;
  float noise = snoise(noiseCoord * 0.15) * 0.5
              + snoise(noiseCoord * 0.3) * 0.25
              + snoise(noiseCoord * 0.6) * 0.125;
  float grainMask = 1.0 - u_grain * noise * 0.3;
  out_fragColor = vec4(color.rgb * grainMask, color.a * grainMask);
}
```

Export a `buildGrainPostSrc()` helper that concatenates version + precision + SIMPLEX_NOISE_GLSL + grain body.

3. **EDGE_DARKEN_POST_FRAG_SRC** -- Fragment shader post-pass for ink edge darkening. Reads accumulated texture, darkens pixels where alpha > threshold, simulating ink pooling at overlaps:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform sampler2D u_input;
uniform float u_edgeDarken;  // 0-1 strength
out vec4 out_fragColor;
void main() {
  vec4 color = texture(u_input, v_texCoord);
  if (color.a < 0.001) { out_fragColor = color; return; }
  float darkenFactor = smoothstep(0.4, 0.9, color.a) * u_edgeDarken * 0.4;
  out_fragColor = vec4(color.rgb * (1.0 - darkenFactor), color.a);
}
```

4. **BLEED_POST_FRAG_SRC** -- Fragment shader post-pass for watercolor bleed. Applies a directional spread by sampling neighbors with noise-based offsets:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform sampler2D u_input;
uniform float u_bleed;       // 0-1 bleed strength
uniform vec2 u_resolution;
out vec4 out_fragColor;

// simplex noise functions prepended at compile time

void main() {
  vec4 center = texture(u_input, v_texCoord);
  vec2 texel = 1.0 / u_resolution;
  float bleedRadius = u_bleed * 4.0;

  // Sample neighbors with noise-offset for organic edge
  float angle = snoise(gl_FragCoord.xy * 0.1) * 3.14159;
  vec2 offset = vec2(cos(angle), sin(angle)) * texel * bleedRadius;

  vec4 s1 = texture(u_input, v_texCoord + offset);
  vec4 s2 = texture(u_input, v_texCoord - offset);
  vec4 s3 = texture(u_input, v_texCoord + offset.yx * vec2(1.0, -1.0));
  vec4 s4 = texture(u_input, v_texCoord - offset.yx * vec2(1.0, -1.0));

  vec4 blurred = (center * 2.0 + s1 + s2 + s3 + s4) / 6.0;

  // Only bleed where there is paint (alpha > 0)
  float bleedMask = smoothstep(0.0, 0.1, max(center.a, max(max(s1.a, s2.a), max(s3.a, s4.a))));
  out_fragColor = mix(center, blurred, u_bleed * bleedMask * 0.6);
}
```

Export a `buildBleedPostSrc()` helper that concatenates version + precision + SIMPLEX_NOISE_GLSL + bleed body.

5. **SCATTER_OFFSET_FRAG_SRC** -- Fragment shader for charcoal scatter. Offsets the stamp sampling position by noise:
```glsl
#version 300 es
precision highp float;
in vec2 v_texCoord;
uniform vec4 u_color;
uniform float u_hardness;
uniform float u_scatter;     // 0-1 scatter amount
out vec4 out_fragColor;

// simplex noise prepended

void main() {
  vec2 uv = v_texCoord;
  float noiseX = snoise(gl_FragCoord.xy * 0.2) * u_scatter * 0.1;
  float noiseY = snoise(gl_FragCoord.xy * 0.2 + 100.0) * u_scatter * 0.1;
  uv += vec2(noiseX, noiseY);
  float dist = length(uv - 0.5) * 2.0;
  float alpha = 1.0 - smoothstep(u_hardness, 1.0, dist);
  // Add grain texture to stamp itself
  float grain = snoise(gl_FragCoord.xy * 0.4) * 0.3 + 0.85;
  out_fragColor = vec4(u_color.rgb, u_color.a * alpha * grain);
}
```

Export a `buildScatterStampSrc()` helper that concatenates version + precision + SIMPLEX_NOISE_GLSL + scatter body.
