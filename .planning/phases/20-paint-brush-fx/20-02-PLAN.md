```typescript
/**
 * Render paint frame with solid background (per D-11).
 * Fills the canvas with the paint background color first, then renders all elements.
 * This is the primary entry point for paint layer compositing.
 */
export function renderPaintFrameWithBg(
  ctx: CanvasRenderingContext2D,
  frame: PaintFrame,
  width: number,
  height: number,
): void {
  // Solid paint background per D-11 (default white, user-configurable)
  const bgColor = paintStore.paintBgColor.peek();
  ctx.save();
  ctx.fillStyle = bgColor;
  ctx.fillRect(0, 0, width, height);
  ctx.restore();

  // Render elements on top of solid background
  renderPaintFrame(ctx, frame, width, height);
}
```

3. Update the `flushStyledStrokes()` function to check for per-stroke cached canvases. Replace the current implementation with one that handles both cached and uncached strokes while preserving z-order:

```typescript
function flushStyledStrokes(
  ctx: CanvasRenderingContext2D,
  strokes: PaintStroke[],
  width: number,
  height: number,
): void {
  // Separate cached vs uncached strokes
  const uncached: PaintStroke[] = [];
  const hasCached = strokes.some(s => s.fxCachedCanvas && s.fxState === 'fx-applied');

  if (!hasCached) {
    // Fast path: no cached strokes, render all via p5.brush batch (existing behavior)
    try {
      const fxCanvas = renderStyledStrokes(strokes, width, height);
      if (fxCanvas) {
        ctx.drawImage(fxCanvas, 0, 0);
        return;
      }
    } catch (e) {
      console.warn('[paintRenderer] brush FX render failed, falling back to flat:', e);
    }
    for (const stroke of strokes) {
      renderStroke(ctx, stroke);
    }
    return;
  }

  // Mixed path: some strokes have cached canvases, others need p5.brush
  // Must composite in z-order: batch consecutive uncached strokes for p5.brush,
  // draw cached strokes individually via drawImage()
  const batch: PaintStroke[] = [];

  const flushBatch = () => {
    if (batch.length === 0) return;
    try {
      const fxCanvas = renderStyledStrokes(batch, width, height);
      if (fxCanvas) {
        ctx.drawImage(fxCanvas, 0, 0);
      }
    } catch (e) {
      console.warn('[paintRenderer] brush FX batch failed, falling back to flat:', e);
      for (const s of batch) renderStroke(ctx, s);
    }
    batch.length = 0;
  };

  for (const stroke of strokes) {
    if (stroke.fxCachedCanvas && stroke.fxState === 'fx-applied') {
      // Flush any pending uncached batch first
      flushBatch();
      // Draw cached raster (per D-03: composited via drawImage)
      ctx.save();
      ctx.globalAlpha = stroke.opacity;
      ctx.drawImage(stroke.fxCachedCanvas, 0, 0);
      ctx.restore();
    } else {
      batch.push(stroke);
    }
  }
  // Flush remaining uncached strokes
  flushBatch();
}
```
