**Add new export `renderSingleStroke()`:**
```typescript
/**
 * Render a single PaintStroke via p5.brush and return its own HTMLCanvasElement.
 * The returned canvas is the cached raster for this stroke (per D-03).
 * Returns null if rendering fails or if the stroke is flat/eraser.
 */
export function renderSingleStroke(
  stroke: PaintStroke,
  width: number,
  height: number,
): HTMLCanvasElement | null {
  if (stroke.tool !== 'brush' || !stroke.brushStyle || stroke.brushStyle === 'flat') {
    return null;
  }

  if (!ensureInitialized(width, height)) return null;

  const halfW = width / 2;
  const halfH = height / 2;

  brush.clear();
  if (_gl) {
    _gl.clearColor(0, 0, 0, 0);
    _gl.clear(_gl.COLOR_BUFFER_BIT);
  }

  const brushName = STYLE_MAP[stroke.brushStyle] || 'marker';
  const params = stroke.brushParams ?? {};
  const pts = preparePoints(stroke.points, halfW, halfH, 20);

  brush.seed(strokeSeed(stroke.id));

  if ((params.fieldStrength ?? 0) > 0.01) {
    brush.field('curved');
    brush.wiggle(params.fieldStrength!);
  }

  if (stroke.brushStyle === 'watercolor') {
    renderWatercolorStroke(stroke, pts);
  } else {
    brush.set(brushName, stroke.color, compensatedWeight(brushName, stroke.size));
    if (pts.length >= 2) {
      brush.spline(pts, 0.5);
    }
  }

  brush.noField();
  brush.render();
  if (_gl) _gl.finish();

  // Copy to a new canvas for per-stroke cache (don't return the shared _canvas)
  const cached = document.createElement('canvas');
  cached.width = width;
  cached.height = height;
  const cCtx = cached.getContext('2d');
  if (cCtx && _canvas) {
    cCtx.drawImage(_canvas, 0, 0);
  }
  return cached;
}
```
