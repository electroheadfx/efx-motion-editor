Create Application/src/lib/brushFlowField.ts implementing a 2D angle grid flow field system.

**FlowField interface:**
```typescript
export interface FlowField {
  grid: Float32Array;  // angles in radians
  cols: number;
  rows: number;
  cellSize: number;
}
```

**createFlowField(width: number, height: number, cellSize?: number): FlowField**
Export. Creates a flow field grid covering the given dimensions.
- Default cellSize = 20 pixels
- cols = Math.ceil(width / cellSize)
- rows = Math.ceil(height / cellSize)
- Fill grid with Perlin-like noise angles using a simple 2D hash-based noise:
```typescript
function hash2D(x: number, y: number): number {
  // Simple hash for deterministic noise
  let h = x * 374761393 + y * 668265263;
  h = (h ^ (h >> 13)) * 1274126177;
  h = h ^ (h >> 16);
  return (h & 0x7fffffff) / 0x7fffffff; // 0-1
}

function noise2D(x: number, y: number): number {
  const ix = Math.floor(x);
  const iy = Math.floor(y);
  const fx = x - ix;
  const fy = y - iy;
  // Smoothstep
  const sx = fx * fx * (3 - 2 * fx);
  const sy = fy * fy * (3 - 2 * fy);
  // Four corners
  const a = hash2D(ix, iy);
  const b = hash2D(ix + 1, iy);
  const c = hash2D(ix, iy + 1);
  const d = hash2D(ix + 1, iy + 1);
  // Bilinear interpolation
  return a * (1-sx)*(1-sy) + b * sx*(1-sy) + c * (1-sx)*sy + d * sx*sy;
}
```
- For each cell: `grid[y * cols + x] = noise2D(x * 0.08, y * 0.08) * Math.PI * 2`
- The 0.08 scale factor gives gentle, large-scale flow patterns

**sampleField(field: FlowField, x: number, y: number): number**
Export. Returns the angle at a given pixel position by looking up the nearest grid cell.
```typescript
const col = Math.min(Math.floor(x / field.cellSize), field.cols - 1);
const row = Math.min(Math.floor(y / field.cellSize), field.rows - 1);
return field.grid[Math.max(0, row) * field.cols + Math.max(0, col)];
```

**applyFlowField(points: Array<{x: number, y: number}>, field: FlowField, strength: number): Array<{x: number, y: number}>**
Export. Displaces each point along the flow field direction by `strength * cellSize * 0.5` pixels.
```typescript
return points.map(p => {
  const angle = sampleField(field, p.x, p.y);
  const displacement = strength * field.cellSize * 0.5;
  return {
    x: p.x + Math.cos(angle) * displacement,
    y: p.y + Math.sin(angle) * displacement,
  };
});
```

Also export a module-level cached flow field:

```typescript
let _cachedField: FlowField | null = null;
let _cachedW = 0;
let _cachedH = 0;

export function getFlowField(w: number, h: number): FlowField {
  if (_cachedField && _cachedW === w && _cachedH === h) return _cachedField;
  _cachedField = createFlowField(w, h);
  _cachedW = w;
  _cachedH = h;
  return _cachedField;
}
```

```typescript
function aggregatePostEffectParams(strokes: PaintStroke[]): {
  grain: number; edgeDarken: number; bleed: number;
} {
  let grain = 0, edgeDarken = 0, bleed = 0;
  for (const s of strokes) {
    const config = getStyleConfig(s.brushStyle ?? 'ink');
    const p = s.brushParams ?? {};
    if (config.postEffects.includes('grain')) grain = Math.max(grain, p.grain ?? 0);
    if (config.postEffects.includes('edgeDarken')) edgeDarken = Math.max(edgeDarken, p.edgeDarken ?? 0);
    if (config.postEffects.includes('bleed')) bleed = Math.max(bleed, p.bleed ?? 0);
  }
  return { grain, edgeDarken, bleed };
}
```
