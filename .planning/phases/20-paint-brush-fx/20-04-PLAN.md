```typescript
flattenFrame(layerId: string, frame: number): void {
  const paintFrame = this.getFrame(layerId, frame);
  if (!paintFrame || paintFrame.elements.length === 0) return;

  const projW = projectStore.width.peek();
  const projH = projectStore.height.peek();

  // Ensure all FX-applied strokes have cached canvases
  for (const el of paintFrame.elements) {
    if (el.tool !== 'brush') continue;
    const stroke = el as PaintStroke;
    if (stroke.brushStyle && stroke.brushStyle !== 'flat' &&
        stroke.fxState === 'fx-applied' && !stroke.fxCachedCanvas) {
      const cached = renderSingleStroke(stroke, projW, projH);
      if (cached) stroke.fxCachedCanvas = cached;
    }
  }

  // Create merged canvas
  const merged = document.createElement('canvas');
  merged.width = projW;
  merged.height = projH;
  const mCtx = merged.getContext('2d')!;

  // Solid background
  const bgColor = paintBgColor.peek();
  mCtx.fillStyle = bgColor;
  mCtx.fillRect(0, 0, projW, projH);

  // Composite all elements in order
  for (const el of paintFrame.elements) {
    if (el.tool === 'brush' || el.tool === 'eraser') {
      const stroke = el as PaintStroke;
      if (stroke.fxCachedCanvas && (stroke.fxState === 'fx-applied' || stroke.fxState === 'flattened')) {
        mCtx.save();
        mCtx.globalAlpha = stroke.opacity;
        mCtx.drawImage(stroke.fxCachedCanvas, 0, 0);
        mCtx.restore();
      } else {
        const path = strokeToPath(stroke.points, stroke.size, stroke.options);
        if (path) {
          mCtx.save();
          if (stroke.tool === 'eraser') {
            mCtx.globalCompositeOperation = 'destination-out';
            mCtx.globalAlpha = 1;
            mCtx.fillStyle = '#000000';
          } else {
            mCtx.globalAlpha = stroke.opacity;
            mCtx.fillStyle = stroke.color;
          }
          mCtx.fill(path);
          mCtx.restore();
        }
      }
    }
  }

  // Store the flattened raster
  _flattenedFrames.set(`${layerId}:${frame}`, merged);

  // Mark all strokes as flattened (original vector data preserved for rollback per D-10)
  for (const el of paintFrame.elements) {
    if (el.tool === 'brush') {
      (el as PaintStroke).fxState = 'flattened';
    }
  }

  this.markDirty(layerId, frame);
  paintVersion.value++;
  _markProjectDirty?.();
},

getFlattenedCanvas(layerId: string, frame: number): HTMLCanvasElement | null {
  return _flattenedFrames.get(`${layerId}:${frame}`) ?? null;
},

unflattenFrame(layerId: string, frame: number): void {
  _flattenedFrames.delete(`${layerId}:${frame}`);
  const paintFrame = this.getFrame(layerId, frame);
  if (!paintFrame) return;
  const projW = projectStore.width.peek();
  const projH = projectStore.height.peek();
  for (const el of paintFrame.elements) {
    if (el.tool !== 'brush') continue;
    const stroke = el as PaintStroke;
    if (stroke.brushStyle && stroke.brushStyle !== 'flat') {
      const cached = renderSingleStroke(stroke, projW, projH);
      stroke.fxCachedCanvas = cached ?? undefined;
      stroke.fxState = 'fx-applied';
    } else {
      stroke.fxState = 'flat';
      stroke.fxCachedCanvas = undefined;
    }
  }
  this.markDirty(layerId, frame);
  paintVersion.value++;
},
```

**2. Update paintRenderer.ts renderPaintFrameWithBg for flatten fast path:**

Update the signature to accept optional layerId and frameNum:
```typescript
export function renderPaintFrameWithBg(
  ctx: CanvasRenderingContext2D,
  frame: PaintFrame,
  width: number,
  height: number,
  layerId?: string,
  frameNum?: number,
): void {
  // Fast path: flattened frame has single pre-composited raster (per D-17)
  if (layerId !== undefined && frameNum !== undefined) {
    const flattened = paintStore.getFlattenedCanvas(layerId, frameNum);
    if (flattened) {
      ctx.drawImage(flattened, 0, 0);
      return;
    }
  }

  // Normal path: solid bg + element-by-element rendering
  const bgColor = paintStore.paintBgColor.peek();
  ctx.save();
  ctx.fillStyle = bgColor;
  ctx.fillRect(0, 0, width, height);
  ctx.restore();

  renderPaintFrame(ctx, frame, width, height);
}
```

import {describe, it, expect} from 'vitest';
import type {PaintStroke, PaintFrame, BrushFxParams} from '../types/paint';

describe('paintPersistence brush FX round-trip', () => {
  const makeStroke = (overrides: Partial<PaintStroke> = {}): PaintStroke => ({
    id: 'test-stroke',
    tool: 'brush',
    points: [[10, 20, 0.5], [30, 40, 0.7]],
    color: '#FF0000',
    opacity: 1,
    size: 10,
    options: {
      thinning: 0.5, smoothing: 0.5, streamline: 0.5,
      simulatePressure: true, pressureEasing: 'linear',
      pressureCurve: 2, taperStart: 0, taperEnd: 0, tiltInfluence: 0.3,
    },
    ...overrides,
  });

  it('PaintStroke with brushStyle serializes to JSON correctly', () => {
    const stroke = makeStroke({brushStyle: 'watercolor'});
    const json = JSON.stringify(stroke);
    const parsed = JSON.parse(json);
    expect(parsed.brushStyle).toBe('watercolor');
  });

  it('PaintStroke with brushParams serializes all FX param values', () => {
    const params: BrushFxParams = {bleed: 0.6, grain: 0.4, fieldStrength: 0.3};
    const stroke = makeStroke({brushStyle: 'watercolor', brushParams: params});
    const json = JSON.stringify(stroke);
    const parsed = JSON.parse(json);
    expect(parsed.brushParams.bleed).toBe(0.6);
    expect(parsed.brushParams.grain).toBe(0.4);
    expect(parsed.brushParams.fieldStrength).toBe(0.3);
  });

  it('PaintStroke without brushStyle deserializes with undefined (backward compat)', () => {
    const stroke = makeStroke();
    delete (stroke as any).brushStyle;
    const json = JSON.stringify(stroke);
    const parsed = JSON.parse(json);
    expect(parsed.brushStyle).toBeUndefined();
  });

  it('Old sidecar JSON without brushStyle/brushParams loads without error', () => {
    const oldJson = JSON.stringify({
      elements: [{
        id: 'old-stroke', tool: 'brush',
        points: [[0, 0, 0.5]], color: '#000', opacity: 1, size: 5,
        options: {thinning: 0.5, smoothing: 0.5, streamline: 0.5,
          simulatePressure: true, pressureEasing: 'linear',
          pressureCurve: 2, taperStart: 0, taperEnd: 0, tiltInfluence: 0.3},
      }],
    });
    const parsed: PaintFrame = JSON.parse(oldJson);
    expect(parsed.elements).toHaveLength(1);
    const s = parsed.elements[0] as PaintStroke;
    expect(s.brushStyle).toBeUndefined();
    expect(s.brushParams).toBeUndefined();
    expect(s.fxState).toBeUndefined();
  });

  it('Round-trip: serialize then deserialize preserves brushStyle and brushParams', () => {
    const frame: PaintFrame = {
      elements: [
        makeStroke({id: 's1', brushStyle: 'ink', brushParams: {edgeDarken: 0.7, fieldStrength: 0.15}, fxState: 'fx-applied'}),
        makeStroke({id: 's2', brushStyle: 'charcoal', brushParams: {grain: 0.6, scatter: 0.4}, fxState: 'fx-applied'}),
      ],
    };
    const json = JSON.stringify(frame);
    const parsed: PaintFrame = JSON.parse(json);
    const s1 = parsed.elements[0] as PaintStroke;
    const s2 = parsed.elements[1] as PaintStroke;
    expect(s1.brushStyle).toBe('ink');
    expect(s1.brushParams?.edgeDarken).toBe(0.7);
    expect(s1.fxState).toBe('fx-applied');
    expect(s2.brushStyle).toBe('charcoal');
    expect(s2.brushParams?.grain).toBe(0.6);
    expect(s2.fxState).toBe('fx-applied');
    expect(s1.fxCachedCanvas).toBeUndefined();
  });
});
