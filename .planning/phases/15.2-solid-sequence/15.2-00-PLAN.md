Follow the pattern from projectStore.test.ts: import from vitest, import the store, use beforeEach to reset state.

Write tests for the 3 new methods that Plan 01 will add. These tests WILL FAIL until Plan 01 implements the methods -- that is expected (RED phase). The tests should import `sequenceStore` and exercise methods via signal reads.

Test structure:

```typescript
import {describe, it, expect, beforeEach} from 'vitest';
import {sequenceStore} from './sequenceStore';

/** Helper: create a minimal content sequence and add it to the store */
function seedSequence(): string {
  sequenceStore.addSequence('content');
  const seq = sequenceStore.sequences.value[0];
  return seq.id;
}

describe('sequenceStore solid/transparent', () => {
  beforeEach(() => {
    // Reset sequences to empty state
    sequenceStore.sequences.value = [];
    sequenceStore.activeSequenceId.value = null;
    sequenceStore.selectedKeyPhotoId.value = null;
  });

  describe('addKeySolid (SOLID-01, D-04, D-06)', () => {
    it('adds a key solid with default black color to the sequence', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId);
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kp = seq.keyPhotos[seq.keyPhotos.length - 1];
      expect(kp.solidColor).toBe('#000000');
      expect(kp.imageId).toBe('');
      expect(kp.holdFrames).toBe(4);
      expect(kp.isTransparent).toBeUndefined();
    });

    it('adds a key solid with custom color', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId, '#FF0000', 8);
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kp = seq.keyPhotos[seq.keyPhotos.length - 1];
      expect(kp.solidColor).toBe('#FF0000');
      expect(kp.holdFrames).toBe(8);
    });
  });

  describe('updateKeySolidColor (SOLID-04, D-14)', () => {
    it('changes the solidColor of an existing key solid', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId, '#000000');
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kpId = seq.keyPhotos[seq.keyPhotos.length - 1].id;

      sequenceStore.updateKeySolidColor(seqId, kpId, '#00FF00');
      const updated = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      expect(updated.keyPhotos.find(kp => kp.id === kpId)!.solidColor).toBe('#00FF00');
    });
  });

  describe('toggleKeyEntryTransparent (SOLID-03, D-11, D-12)', () => {
    it('toggles a key solid to transparent mode', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId, '#FF0000');
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kpId = seq.keyPhotos[seq.keyPhotos.length - 1].id;

      sequenceStore.toggleKeyEntryTransparent(seqId, kpId);
      const toggled = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kp = toggled.keyPhotos.find(k => k.id === kpId)!;
      expect(kp.isTransparent).toBe(true);
    });

    it('round-trips transparent back to solid, restoring the original solidColor', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId, '#FF0000');
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kpId = seq.keyPhotos[seq.keyPhotos.length - 1].id;

      // Toggle to transparent
      sequenceStore.toggleKeyEntryTransparent(seqId, kpId);
      // Toggle back to solid
      sequenceStore.toggleKeyEntryTransparent(seqId, kpId);

      const restored = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kp = restored.keyPhotos.find(k => k.id === kpId)!;
      expect(kp.isTransparent).toBeUndefined();
      expect(kp.solidColor).toBe('#FF0000');
    });

    it('defaults to #000000 when toggling from transparent to solid with no prior color', () => {
      const seqId = seedSequence();
      sequenceStore.addKeySolid(seqId, '#000000');
      const seq = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kpId = seq.keyPhotos[seq.keyPhotos.length - 1].id;

      // Modify to remove solidColor then toggle
      sequenceStore.toggleKeyEntryTransparent(seqId, kpId);
      sequenceStore.toggleKeyEntryTransparent(seqId, kpId);

      const restored = sequenceStore.sequences.value.find(s => s.id === seqId)!;
      const kp = restored.keyPhotos.find(k => k.id === kpId)!;
      expect(kp.solidColor).toBe('#000000');
    });
  });
});
```

**B) Create `Application/src/lib/frameMap.test.ts`**

This test validates that frameMap produces FrameEntry with solidColor/isTransparent when the source keyPhotos have those fields. Since frameMap is a `computed` signal that reads from `sequenceStore.sequences`, the test sets up sequences with solid key photos and reads the computed.

```typescript
import {describe, it, expect, beforeEach} from 'vitest';
import {sequenceStore} from '../stores/sequenceStore';
import {frameMap} from './frameMap';

describe('frameMap solid/transparent entries', () => {
  beforeEach(() => {
    sequenceStore.sequences.value = [];
    sequenceStore.activeSequenceId.value = null;
  });

  it('produces FrameEntry with solidColor for key solid entries', () => {
    sequenceStore.sequences.value = [{
      id: 'seq-1',
      name: 'Test',
      kind: 'content',
      fps: 24,
      keyPhotos: [
        { id: 'kp-1', imageId: '', holdFrames: 2, solidColor: '#FF0000' },
      ],
      layers: [],
      transitions: [],
    }];

    const entries = frameMap.value;
    expect(entries).toHaveLength(2);
    expect(entries[0].solidColor).toBe('#FF0000');
    expect(entries[0].imageId).toBe('');
    expect(entries[1].solidColor).toBe('#FF0000');
  });

  it('produces FrameEntry with isTransparent for transparent entries', () => {
    sequenceStore.sequences.value = [{
      id: 'seq-1',
      name: 'Test',
      kind: 'content',
      fps: 24,
      keyPhotos: [
        { id: 'kp-1', imageId: '', holdFrames: 1, solidColor: '#000000', isTransparent: true },
      ],
      layers: [],
      transitions: [],
    }];

    const entries = frameMap.value;
    expect(entries).toHaveLength(1);
    expect(entries[0].isTransparent).toBe(true);
  });

  it('produces FrameEntry without solidColor for regular key photos', () => {
    sequenceStore.sequences.value = [{
      id: 'seq-1',
      name: 'Test',
      kind: 'content',
      fps: 24,
      keyPhotos: [
        { id: 'kp-1', imageId: 'img-1', holdFrames: 1 },
      ],
      layers: [],
      transitions: [],
    }];

    const entries = frameMap.value;
    expect(entries).toHaveLength(1);
    expect(entries[0].solidColor).toBeUndefined();
    expect(entries[0].isTransparent).toBeUndefined();
    expect(entries[0].imageId).toBe('img-1');
  });

  it('interleaves solid and photo FrameEntry correctly', () => {
    sequenceStore.sequences.value = [{
      id: 'seq-1',
      name: 'Test',
      kind: 'content',
      fps: 24,
      keyPhotos: [
        { id: 'kp-1', imageId: 'img-1', holdFrames: 1 },
        { id: 'kp-2', imageId: '', holdFrames: 2, solidColor: '#0000FF' },
        { id: 'kp-3', imageId: 'img-2', holdFrames: 1 },
      ],
      layers: [],
      transitions: [],
    }];

    const entries = frameMap.value;
    expect(entries).toHaveLength(4);
    expect(entries[0].imageId).toBe('img-1');
    expect(entries[0].solidColor).toBeUndefined();
    expect(entries[1].solidColor).toBe('#0000FF');
    expect(entries[2].solidColor).toBe('#0000FF');
    expect(entries[3].imageId).toBe('img-2');
    expect(entries[3].solidColor).toBeUndefined();
  });
});
```

NOTE: The Sequence type may need `as any` casting to satisfy TypeScript if the KeyPhoto type does not yet have solidColor/isTransparent fields. Use inline type assertion where needed:
```typescript
keyPhotos: [
  { id: 'kp-1', imageId: '', holdFrames: 2, solidColor: '#FF0000' } as any,
],
```
This ensures the test file compiles pre-Plan-01. After Plan 01 extends the type, the `as any` can be removed (but is not required to be removed for tests to pass).
