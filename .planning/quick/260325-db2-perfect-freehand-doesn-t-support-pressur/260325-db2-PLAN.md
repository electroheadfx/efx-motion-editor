<objective>
Add proper tablet pen support (pressure, tilt, coalesced events) to the paint system.

Purpose: The perfect-freehand library already accepts pressure as the 3rd value in `[x, y, pressure]` point arrays and uses it to modulate stroke width via the `thinning` parameter. However, the current implementation has several gaps: (1) no smart detection of pen vs mouse — `simulatePressure` is hardcoded `false`, (2) no coalesced event handling — tablet pens report at 200-240Hz but browser pointermove fires at ~60Hz, losing intermediate points, (3) no tilt/angle capture from tablet pens, (4) the `easing` and `start`/`end` taper options of getStroke are unused.

Output: Tablet-aware paint strokes with real pressure, full-resolution coalesced input, tilt modulation, and automatic input device detection.

```typescript
function getCoalescedPoints(
  e: PointerEvent,
  containerRef: RefObject<HTMLDivElement>,
): Array<{x: number; y: number; pressure: number; tiltX: number; tiltY: number}> {
  const coalesced = (e as any).getCoalescedEvents?.() as PointerEvent[] | undefined;
  const events = coalesced && coalesced.length > 0 ? coalesced : [e];
  return events.map(ev => {
    const pt = getProjectPointFromEvent(ev, containerRef);
    return {
      ...pt,
      pressure: ev.pressure,
      tiltX: ev.tiltX || 0,
      tiltY: ev.tiltY || 0,
    };
  });
}
```

```typescript
// Pressure easing function lookup
const PRESSURE_EASINGS: Record<string, (p: number) => number> = {
  linear: (p: number) => p,
  gentle: (p: number) => p * (2 - p),
  firm: (p: number) => p * p,
};

export function strokeToPath(
  points: [number, number, number][],
  size: number,
  options: PaintStrokeOptions,
): Path2D | null {
  const easing = PRESSURE_EASINGS[options.pressureEasing] || PRESSURE_EASINGS.linear;

  const outline = getStroke(points, {
    size,
    thinning: options.thinning,
    smoothing: options.smoothing,
    streamline: options.streamline,
    simulatePressure: options.simulatePressure,
    easing,
    start: options.taperStart > 0 ? {
      taper: options.taperStart === -1 ? true : options.taperStart,
      cap: true,
    } : { cap: true },
    end: options.taperEnd > 0 ? {
      taper: options.taperEnd === -1 ? true : options.taperEnd,
      cap: true,
    } : { cap: true },
    last: true,
  });

  if (outline.length < 2) return null;

  const path = new Path2D();
  path.moveTo(outline[0][0], outline[0][1]);
  for (let i = 1; i < outline.length; i++) {
    path.lineTo(outline[i][0], outline[i][1]);
  }
  path.closePath();
  return path;
}
```
